How to Solve the Program

The program calculates the new salary of the employee as per the following specifications.

Code For The Program

import java.util.*;

public class Grade_Revision

{

String name;

int bas;

int expn;

double inc;

double nbas;

Grade_Revision()

{

name="";

bas=0;

expn=0;

inc=0.0;

nbas=0.0;

}

void accept()

{

Scanner sc=new Scanner(System.in);

System.out.println("Enter the name");

name=sc.next();

System.out.println("Enter the basic salary");

bas=sc.nextInt();

System.out.println("Enter the experience");

expn=sc.nextInt();

}

void increment()

{

if(expn<3)

inc=1000.0+10.0/100.0*bas;

else if(expn>=3&&expn<5)

inc=3000.0+12.0/100.0*bas;

else if(expn>=5&&expn<10)

inc=5000.0+15.0/100.0*bas;

else

inc=8000.0+20.0/100.0*bas;

nbas=bas+inc;

}

void display()

{

System.out.println("The name is "+name);

System.out.println("The experience is"+expn);

System.out.println("The new salary is"+nbas);

}

static void main()

{

Grade_Revision sc=new Grade_Revision();

sc.accept();

sc.increment();

sc.display();

}

}
