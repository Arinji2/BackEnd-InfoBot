ATM-Bank

import java.util.*;

public class Bank {

public static void main() {

Scanner in=new Scanner(System.in);

int choice,k=0;

double balance,deposit,withdraw;

System.out.println("Enter your current balance");

balance=in.nextDouble();

while(true){

if(k>=3) {

System.out.println("Do you want to exit? Type 0");

}

System.out.println("Enter your choice");

choice=in.nextInt();

switch(choice){

case 1:

System.out.println("Enter the money to be deposited");

deposit=in.nextDouble();

balance+=deposit;

System.out.println("Current Balance= "+balance);

break;

case 2:

System.out.println("Enter the money to be withdrawn");

withdraw=in.nextDouble();

if(balance>=withdraw) {

balance-=withdraw;

System.out.println("Current Balance= "+balance);

}

else {

System.out.println("Balance not sufficient");

}

break;

case 3:

System.out.println("Current Balance= "+balance);

break;

case 0:

System.out.println("Program terminates");

System.exit(0);

break;

default:

System.out.println("Invalid choice");

System.exit(0);

}

k++;

}

}

}

The choice is inputted and goes to the switch block. After getting out of the satiable condition, it goes back to the start(loop) and continues till the user types '0' to exit.
